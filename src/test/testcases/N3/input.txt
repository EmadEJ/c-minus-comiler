int GCD(int a, int b) {
	while (1) {
		if (a < b) {
			b = b - a;
		}
		else {
			a = a - b;
		}
		if (a < 1) {
			return b;
		}
		else {
			a = a;
		}
		if (b < 1) {
			return a;
		}
		else {
			b = b;
		}
	}
}

void main(void) {
  	output(GCD(1, 10));
  	output(GCD(420, 100));
  	output(GCD(31, 31));
  	output(GCD(1382, 2003));
}
